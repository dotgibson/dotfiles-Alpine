# dotfiles-Alpine/install/packages.txt
# apk package names for the modern terminal stack. One per line; # comments ok.
# Tools NOT packaged (starship) or built from source on musl (yazi, tree-sitter)
# are handled by bootstrap.sh — keep them OUT of here.
# Most modern tools live in the `community` repo — make sure it's enabled in
# /etc/apk/repositories (it is by default on Alpine 3.x + edge).

# ── shell + multiplexer ───────────────────────────────────────
bash           # bootstrap.sh is a bash script; a fresh Alpine ships only busybox ash,
               # so `apk add bash` is a PREREQUISITE to run it (see README) — declared
               # here too so a full provision keeps it installed
zsh            # Alpine's default shell is ash — zsh must be added explicitly
tmux

# ── privilege + shell tooling ─────────────────────────────────
doas           # Alpine's default privilege tool (sudo's lean cousin)
shadow         # provides chsh/usermod so bootstrap can set zsh as login shell

# ── modern CLI core (community repo) ──────────────────────────
eza            # ls
bat            # cat
fd             # find (Alpine packages it as `fd`)
ripgrep        # grep
zoxide         # cd
fzf            # fuzzy finder
delta          # git pager / diffs  (Alpine package is `delta`, NOT git-delta)
btop           # top/htop
tealdeer       # tldr (binary: tldr)
atuin          # shell history (in Alpine repos — no installer needed)

# ── modern CLI extras (core-doctor parity; community repo) ─────
# The rest of core-doctor's modern-CLI set. All in `community` EXCEPT duf/glow,
# which live in `testing` (not enabled on a stable box) → installed via `go install`
# in bootstrap.sh instead of here, so a stable Alpine's `apk add` never fails on them.
# apk_install() (bootstrap.sh) is resilient — a name missing on this box (e.g. `sd`
# if it hasn't landed, or a testing-only tool) is skipped per-package, never fatal.
dust           # du (dust)
procs          # ps
gron           # greppable JSON
gum            # shell-script UI toolkit (charm)
xh             # HTTPie-style HTTP client
doggo          # dig / DNS client (in community — no go install needed)
carapace       # multi-shell completion engine (in community — no go install needed)
sd             # sed-style find & replace (MED-confidence on Alpine; verify: apk search -e sd)
yq-go          # mikefarah Go yq — packaged as `yq-go`, provides `cmd:yq`.
               # Do NOT use plain `yq` (that's the Python kislyuk yq, a different tool).

# ── editor + git workflow ─────────────────────────────────────
neovim
git            # `git subtree` ships with git on Alpine (no separate package)
openssh-client # ssh / ssh-keygen / ssh-agent
lazygit

# ── clipboard — OPTIONAL: pulls X/Wayland libs. On a headless/container Alpine
# you can delete both (Core's `clip` just no-ops without a backend; on WSL it
# uses clip.exe regardless). Keep them if you run a desktop session.
wl-clipboard   # wl-copy / wl-paste (Wayland)
xclip          # X11 fallback

# ── build toolchain (needed to compile yazi + tree-sitter-cli on musl) ────────
build-base     # Alpine's gcc/g++/make/musl-dev meta (replaces Fedora's gcc/gcc-c++/make)
cargo          # rust toolchain (pulls rust)
pkgconf        # some crates probe libs via pkg-config at build time
openssl-dev    # ditto — common cargo build dependency

# ── Neovim provider deps (fix :checkhealth) ───────────────────
python3
py3-pip
py3-pynvim     # pynvim — the Python 3 provider module (Alpine name)

# ── misc useful ───────────────────────────────────────────────
curl
wget
jq
gawk
direnv
iproute2       # full `ip` — busybox's `ip` applet lacks `-brief`, which the `localip`
               # alias (os/alpine.zsh) uses; without this it errors on a lean Alpine
# NOTE: no flatpak here — it isn't idiomatic on Alpine (the lean/container distro).
